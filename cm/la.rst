  enum tokenType{
    INTEGER,             /* 25, 543, 1, 78070*/
    DOUBLE /* NUMBER */,
    IDENTIFIER,          /* x, count, f1, power*/
    IF,                  /* if */
    WHILE,               /* while */
    OPEN_PARANTHESIS,    /* ( */
    CLOSE_PARANTHESIS,   /* ) */
    ASSIGN,              /* = */
    OPEN_CURLY_BRACKET,  /* { */
    CLOSE_CURLY_BRACKET, /* } */
    OPEN_BRACKET,        /* [ */
    CLOSE_BRACKET,       /* ] */
    EQUAL,               /* == */
    LESS_THAN,           /* <  */
    GREATER_THAN,        /* > */
    LESS_THAN_EQUAL,     /* <= */
    GREATER_THAN_EQUAL,  /* >= */
    COMMA,               /* , */
    SEMICOLON,           /* ; */
    END_OF_LINE,
    END_OF_FILE
  }

Regular Expression
=======================
.. container:: 

  #. 0|1|2|3 : 0 or 1 or 2 or 3
  #. [0123] : 0 or 1 or 2 or 3
  #. [0-3] : 0 or 1 or 2 or 3
  #. [-03] : 0 or - or 3
  #. (abc) : Capture group abc
  #. (a|b) : Match a or b
  #. . : Any character, except newline
  #. \\. \\* \\\\ \\t \\n \\r \\-
  #. [0\\-3] : 0 or - or 3
  #. \\w : Word
  #. \\d : digit
  #. ^ `:` start of String
  #. ^abc : Start with abc
  #. $ `:` End of string
  #. abc$ : End with abc
  #. [ \\t] : space or tab
  #. [\\n\\t] : new line or tab

.. container::

  #. a+ ``:`` Match 1 or more a
  #. [0-9]+
  #. a* `:` Match 0 or more a
  #. [a-zA-Z\_][a-zA-Z\_]*
  #. [a-zA-Z\_]+
  #. a? `:` Match 0 or 1 of a
  #. if(def)?
  #. [0-9]+(\\.[0-9]+)?
  #. a{3}   : Match exactly 3 , aaa
  #. a{1,3} : Match between 1 and 3
  #. a{,3}  : Match up to 3
  #. a{3,}  : Match 3 or more
  #. [^abc] : Any character except a, b or c
  #. [^\\n] : . ,  scanf("[^\\n]", st);
  #. [a^c] : one of: a ^ c
  #. a/b `:` a when followed by b
  #. (abc)/(xyz) : abcsldsxyzss , abcsldsxzss ×
  #. (https?|ftp):/(/[^/\\r\\n]+)+/?
